2. Led Chasing 
#include <reg51.h> 
 void msdelay(unsigned int); 
 void main (void){ 
  unsigned int z; 
  while(1){ 
   P1=0x80; 
   for(z=0;z<9;z++){ 
    msdelay(50000); 
    P1=P1>>1; 
   } 
  } 
 }

3. Display A to Z Ascii Values

    msdelay(5000);
